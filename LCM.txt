#include <stdio.h>
int main() {
    int a, b, lcm;
    printf("Enter two integer numbers:\n");
    scanf("%d %d", &a, &b);
    int max=(a>b)?a:b;
    for (int i=max; i<=a*b; i++) {
        if (i%a==0&&i%b==0) {
            lcm = i;
            break;
        }
    }
    printf("LCM of %d and %d is %d\n", a, b, lcm);
    return 0;
}
