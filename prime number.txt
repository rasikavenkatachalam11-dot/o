#include <stdio.h>
int is_prime(int n) {
    if (n <= 1) return 0;
    for (int i = 2; i * i <= n; i++) {
        if (n % i == 0) return 0;
    }
    return 1;
}
int count_primes(int N) {
    int count = 0;
    for (int i = 2; i <= N; i++) {
        if (is_prime(i)) {
            count++;
        }
    }
    return count;
}
int main() {
    int N;
    if (scanf("%d", &N) != 1) return 1;
    int count = count_primes(N);
    printf("%d\n", count);
    return 0;
}
