#include <stdio.h>
long long factorial(int n) {
    long long fact = 1;
    for (int i=1; i<= n; ++i) {
        fact*=i;
    }
    return fact;
}
int main() {
    int N, originalN, digit;
    long long sum = 0;
    scanf("%d", &N);
    originalN= N;
    for (; N>0; N/=10) {
        digit=N%10;
        sum+=factorial(digit);
    }
    if (sum == originalN) {
        printf("Yes\n");
    } else {
        printf("No\n");
    }
    return 0;
}
