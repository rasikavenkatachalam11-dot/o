#include <stdio.h>
int main() {
    int n, original, sum = 0, remainder;
    printf("Enter a number: ");
    scanf("%d", &n);
    original = n;
    while (n != 0) {
        remainder = n % 10;
        sum = sum + (remainder * remainder * remainder);
        n = n / 10;
    }
    if (sum == original)
        printf("Armstrong number");
    else
        printf("Not an Armstrong number");
    return 0;
}
