#include<stdio.h>
int main(){
    int marks;
    scanf("%d",&marks);
    switch(marks/10){
        case 9:
        printf("Grade A\n");
        break;
        case 8:
        printf("Grade B\n");
        break;
        case 7:
        printf("Grade C\n");
        break;
        case 6:
        printf("Grade D\n");
        break;
        case 5:
        printf("Grade E\n");
        break;
        case 4:
        printf("Grade F\n");
        break;
        case 3:
        printf("Grade Supplementary\n");
        break;
        case 2:
        printf("Grade Fail\n");
        break;
        
        
    }
    return 0;
}
